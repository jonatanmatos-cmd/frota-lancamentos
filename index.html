// Substitua pelos IDs da sua Planilha e da sua Pasta no Google Drive const SPREADSHEET_ID = "SEU_ID_DA_PLANILHA_AQUI"; const FOLDER_ID = "SEU_ID_DA_PASTA_DO_DRIVE_AQUI"; /** * Configuração para carregar as placas via GET (executada no carregamento da página) */ function doGet(e) { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); // Presume-se que você tenha uma aba chamada "Placas" com a lista na coluna A const sheetPlacas = ss.getSheetByName("Placas"); let listaPlacas = []; if (sheetPlacas) { const valores = sheetPlacas.getRange(2, 1, sheetPlacas.getLastRow() - 1, 1).getValues(); listaPlacas = valores.map(linha => linha[0]).filter(String); } // Caso a aba não exista ou esteja vazia, você pode retornar um fallback ou erro return ContentService.createTextOutput(JSON.stringify({ status: "ok", placas: listaPlacas })).setMimeType(ContentService.MimeType.JSON); } catch (erro) { return ContentService.createTextOutput(JSON.stringify({ status: "erro", message: "Erro ao buscar placas: " + erro.toString() })).setMimeType(ContentService.MimeType.JSON); } } /** * Processa o envio dos dados do formulário (POST) */ function doPost(e) { const lock = LockService.getScriptLock(); try { // Evita concorrência de escritas simultâneas na planilha lock.waitLock(10000); const dados = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SPREADSHEET_ID); // Define a aba de destino com base no tipo de lançamento let nomeAba = dados.tipo === "orcamento" ? "Orçamentos" : "PosServico"; let sheet = ss.getSheetByName(nomeAba); // Se a aba não existir, cria automaticamente com os cabeçalhos if (!sheet) { sheet = ss.insertSheet(nomeAba); if (dados.tipo === "orcamento") { sheet.appendRow(["Data Registro", "Nº Orçamento", "Placa", "Motorista", "Filial", "Tipo Manutenção", "Fornecedor", "Valor Orçado", "Previsão Conclusão", "Link Foto/PDF", "Observações"]); } else { sheet.appendRow(["Data Registro", "Nº Orçamento", "Placa", "Motorista", "Data Conclusão", "Valor Final", "Link NF", "Link NS", "Observações"]); } } const dataAtual = new Date(); const pastaDestino = DriveApp.getFolderById(FOLDER_ID); if (dados.tipo === "orcamento") { // Processa arquivo do Orçamento se houver let urlFoto = ""; if (dados.fotoBase64 && dados.fotoNome) { urlFoto = salvarArquivoNoDrive(dados.fotoBase64, dados.fotoNome, dados.fotoTipo, pastaDestino); } // Adiciona linha na aba de Orçamentos sheet.appendRow([ dataAtual, dados.numeroOrcamento, dados.placa, dados.motorista, dados.filial, dados.tipoManutencao, dados.fornecedor, dados.valor ? parseFloat(dados.valor) : "", dados.dataPrevisao, urlFoto, dados.observacoes ]); } else { // Processa arquivos de Pós-Serviço (NF e NS) se houverem let urlNF = ""; if (dados.nfBase64 && dados.nfNome) { urlNF = salvarArquivoNoDrive(dados.nfBase64, dados.nfNome, dados.nfTipo, pastaDestino); } let urlNS = ""; if (dados.nsBase64 && dados.nsNome) { urlNS = salvarArquivoNoDrive(dados.nsBase64, dados.nsNome, dados.nsTipo, pastaDestino); } // Adiciona linha na aba de Pós-Serviço sheet.appendRow([ dataAtual, dados.numeroOrcamento, dados.placa, dados.motorista, dados.dataConclusao, dados.valorFinal ? parseFloat(dados.valorFinal) : "", urlNF, urlNS, dados.observacoes ]); } return ContentService.createTextOutput(JSON.stringify({ status: "ok", message: "Lançamento realizado com sucesso!" })).setMimeType(ContentService.MimeType.JSON); } catch (erro) { return ContentService.createTextOutput(JSON.stringify({ status: "erro", message: erro.toString() })).setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } } /** * Função auxiliar para decodificar Base64 e salvar no Drive */ function salvarArquivoNoDrive(base64Data, nomeArquivo, tipoMime, pasta) { const bytes = Utilities.base64Decode(base64Data); const blob = Utilities.newBlob(bytes, tipoMime, nomeArquivo); const arquivoCriado = pasta.createFile(blob); // Opcional: Define a permissão para que qualquer pessoa na empresa com o link possa ver o arquivo arquivoCriado.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return arquivoCriado.getUrl(); }
